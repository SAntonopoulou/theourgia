{{- if not .Values.secrets.secretKey }}

WARNING: secrets.secretKey is empty. The backend will refuse to start
until THEOURGIA_SECRET_KEY is set. Generate one with:
  openssl rand -base64 64
{{- end }}
{{- if not .Values.secrets.masterEncryptionKey }}

WARNING: secrets.masterEncryptionKey is empty. The backend will refuse to
start until THEOURGIA_MASTER_ENCRYPTION_KEY is set. Generate one with:
  openssl rand -base64 64
Back this key up outside the cluster — losing it makes all Mode A
(server-side encrypted) content permanently unreadable.
{{- end }}

{{- if .Values.ingress.enabled }}
